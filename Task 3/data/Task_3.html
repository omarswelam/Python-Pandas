def multivariate_gaussian_nonnormalized(x, d=2, mean=mean, covariance=cov):
    """pdf of the multivariate normal distribution."""
    x_m = x - mean
    return np.exp(np.diag(-(np.dot(np.dot(x_m,np.linalg.inv(cov)),x_m.T))))
